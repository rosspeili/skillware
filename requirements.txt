#   pip install skillware                                    # core + all skills on disk
#   pip install "skillware[finance_wallet_screening]"        # per-skill extra (preferred in docs)
#   pip install "skillware[office_pdf_form_filler]"          # per-skill extra with runtime deps
#   pip install "skillware[finance]"                         # whole category
#   pip install "skillware[gemini]"                          # agent SDK
#   pip install "skillware[agents]"                          # all agent SDKs
#   pip install "skillware[all]"                             # all skill runtime deps
#   pip install "skillware[dev]"                             # testing and linting tools
